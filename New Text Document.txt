        }
    }

    private void jdeletewordActionPerformed(java.awt.event.ActionEvent evt) {                                            
        // TODO add your handling code here:
        // delete word
        Component frame = null;
        String str = jTextField4.getText();
        int count = 0;
        for(int i=0;i<arr1.size();i++)
        {
           
            if(arr1.get(i).equals(str))
            {
                count++;
                int check = f.readWord(str, arr1);
                String word = f.readExplain(check, Explains);
                arr2.remove(word);
                arr1.remove(str);
                JOptionPane.showMessageDialog(frame, "Từ đã được xóa!!");
            }
        }
        if(count == 0)
        {
            JOptionPane.showMessageDialog(frame, "Từ không có trong từ điển!!");
        }
        else {
        f.write("", Target);
        f.write("", Explains);
        for(int i=0;i<arr1.size();i++){
            f.writeContinue(arr1.get(i), Target);
            f.writeContinue(arr2.get(i), Explains);
        }
        } 
    }

        // TODO add your handling code here:
        // update word
        Component frame = null;
        String str1 = jTextField5.getText().toLowerCase();
        String str2 = jTextField6.getText().toLowerCase();
        int count = 0;
        for(int i=0;i<arr1.size();i++){
           
            if(arr1.get(i).equals(str1)){
                 count++;
                arr2.set(i, str2);
                JOptionPane.showMessageDialog(frame, "Từ đã được cập nhật!!");

        }
        else {
        //f.write("", Target);
        f.write("", Explains);
        for(int i=0;i<arr1.size();i++){
            //f.writeContinue(arr1.get(i), Target);
            f.writeContinue(arr2.get(i), Explains);
        }
        } 
    }
